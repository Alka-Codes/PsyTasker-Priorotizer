    <!-- Suggestion Area -->
    <h2> Smart Suggestion</h2>
    <p id="suggestion">No tasks yet.</p>

    <script>
        function addTask() {
            let taskName = document.getElementById("taskName").value;
            let loadLevel = document.getElementById("loadLevel").value;

            if (taskName === "") {
                alert("Please enter a task!");
                return;
            }

            let taskList = document.getElementById("taskList");

            let li = document.createElement("li");
            li.textContent = taskName + " [" + loadLevel + "]";
            taskList.appendChild(li);

            generateSuggestion(loadLevel);

            document.getElementById("taskName").value = "";
        }

        function generateSuggestion(loadLevel) {
            let suggestion = document.getElementById("suggestion");

            if (loadLevel === "Low") {
                suggestion.textContent = "Good for quick productivity! Stack multiple low-load tasks.";
            }
            else if (loadLevel === "Medium") {
                suggestion.textContent = "Balanced work. Take short breaks after completion.";
            }
            else {
                suggestion.textContent = "High cognitive load! Work in deep focus sessions (Pomodoro recommended).";
            }
        }
    </script>
